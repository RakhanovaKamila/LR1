Звіт з лабораторної роботи №2

Тема: Перетворення ER-діаграми на схему PostgreSQL
Мета: Написати SQL DDL-інструкції для створення бази даних, визначити типи даних, первинні та зовнішні ключі, додати обмеження та наповнити таблиці тестовими даними.

Опис схеми бази даних

Схема бази даних розроблена для обслуговування ігрового проекту і складається з 11 таблиць, які об'єднані складними реляційними зв'язками. База даних активно використовує користувацькі типи даних ENUM для забезпечення цілісності таких атрибутів, як статуси серверів, класи персонажів, рідкість предметів та ролі гравців.

Фундаментом бази даних виступають незалежні таблиці. Таблиця Account зберігає облікові записи користувачів, маючи обмеження UNIQUE для поштової скриньки та імені користувача, а також перевірку CHECK на мінімальну довжину пароля. Таблиця game_server містить інформацію про сервери, а ItemTemplate є довідником усіх доступних у грі предметів. Кожна з цих таблиць ідентифікується унікальним первинним ключем типу SERIAL.

Наступним рівнем є таблиці, які залежать від базових сутностей через зовнішні ключі (FOREIGN KEY). Таблиця Region логічно прив'язана до конкретного сервера. Таблиця WeaponStas розширює інформацію про предмети-зброю, використовуючи item_id і як первинний, і як зовнішній ключ (зв'язок "один до одного"). Таблиця Quest_System містить ігрові завдання, де обов'язковим атрибутом є посилання на нагороду (item_id з ItemTemplate). Таблиця Game_Structure описує будівлі, які розміщуються в конкретних регіонах.

Найбільшу кількість залежностей має таблиця Game_Character. Вона акумулює посилання на обліковий запис власника, поточний сервер, регіон перебування та будівлю, яка слугує точкою відродження. Для гнучкої взаємодії між персонажами, предметами та світом створено асоціативні таблиці. Таблиця Inventory_Slot реалізує інвентар гравців (зв'язок між персонажами та предметами). Таблиця Character_Quests відстежує прогрес виконання завдань (має складений первинний ключ з ідентифікаторів персонажа та квесту). Таблиця Structure_ownership дозволяє реалізувати систему спільного володіння нерухомістю, зв'язуючи персонажів та будівлі з певними ролями доступу.

SQL DDL: Створення схеми та таблиць

Нижче наведено код для створення типів та таблиць, включаючи всі необхідні обмеження (PRIMARY KEY, FOREIGN KEY, CHECK, DEFAULT).

-- Очищення попередньої схеми
DROP TABLE IF EXISTS Character_Quests CASCADE;
DROP TABLE IF EXISTS Inventory_Slot CASCADE;
DROP TABLE IF EXISTS WeaponStas CASCADE;
DROP TABLE IF EXISTS Quest_System CASCADE;
DROP TABLE IF EXISTS Game_Structure CASCADE;
DROP TABLE IF EXISTS Game_Character CASCADE;
DROP TABLE IF EXISTS Region CASCADE;
DROP TABLE IF EXISTS ItemTemplate CASCADE;
DROP TABLE IF EXISTS game_server CASCADE;
DROP TABLE IF EXISTS Account CASCADE;
DROP TABLE IF EXISTS Structure_ownership CASCADE;

DROP TYPE IF EXISTS quest_status_type CASCADE;
DROP TYPE IF EXISTS class_type CASCADE;
DROP TYPE IF EXISTS structure_type1 CASCADE;
DROP TYPE IF EXISTS bonus_type CASCADE;
DROP TYPE IF EXISTS rarity_type CASCADE;
DROP TYPE IF EXISTS item_category CASCADE;
DROP TYPE IF EXISTS server_status CASCADE;
DROP TYPE IF EXISTS account_status CASCADE;
DROP TYPE IF EXISTS online_status CASCADE;
DROP TYPE IF EXISTS structure_role_type CASCADE;

-- Створення користувацьких типів
CREATE TYPE online_status     AS ENUM ('Online', 'Offline', 'Sleeping');
CREATE TYPE account_status    AS ENUM ('Banned', 'OK', 'Muted', 'Penalized');
CREATE TYPE server_status     AS ENUM ('On', 'Off');
CREATE TYPE item_category     AS ENUM ('weapon', 'resource', 'trash');
CREATE TYPE rarity_type       AS ENUM ('Common', 'Uncommon', 'Rare', 'Legendary');
CREATE TYPE bonus_type        AS ENUM ('Strength', 'Accuracy');
CREATE TYPE structure_type1   AS ENUM ('ruin', 'circus', 'church', 'mine');
CREATE TYPE class_type        AS ENUM ('Mage', 'Warrior', 'Archer', 'Berserker');
CREATE TYPE quest_status_type AS ENUM ('Active', 'Completed', 'Failed');
CREATE TYPE structure_role_type AS ENUM ('Owner', 'Co_owner', 'Guest');

-- Створення таблиць
CREATE TABLE Account (
    account_id        SERIAL PRIMARY KEY,
    email             VARCHAR(255) NOT NULL UNIQUE,
    account_password  VARCHAR(24) CHECK (LENGTH(account_password) >= 8) NOT NULL,
    username          VARCHAR(24) NOT NULL UNIQUE,
    registration_date TIMESTAMP,
    online            online_status NOT NULL,
    status            account_status NOT NULL,
    in_game_time      INTERVAL
);

CREATE TABLE game_server (
    server_id          SERIAL PRIMARY KEY,
    server_name1       VARCHAR(64) NOT NULL,
    ip_address         VARCHAR(45) NOT NULL,
    status             server_status NOT NULL,
    current_characters INT DEFAULT 0 CHECK (current_characters >= 0)
);

CREATE TABLE Region (
    region_id        SERIAL PRIMARY KEY,
    server_id        INT NOT NULL REFERENCES game_server(server_id),
    region_name      VARCHAR(100),
    region_biome     VARCHAR(100),
    region_min_level INT NOT NULL
);

CREATE TABLE Game_Structure (
    structure_id       SERIAL PRIMARY KEY,
    region_id          INT REFERENCES Region(region_id),
    structure_name     VARCHAR(50),
    structure_position DECIMAL(12, 4) NOT NULL DEFAULT 0.0000,
    structure_type     structure_type1 NOT NULL,
    is_buyable         BOOLEAN,
    structure_cost     DECIMAL(10, 2) DEFAULT 0
);

CREATE TABLE Game_Character (
    character_id         SERIAL PRIMARY KEY,
    account_id           INT REFERENCES Account(account_id),
    server_id            INT REFERENCES game_server(server_id),
    region_id            INT REFERENCES Region(region_id),
    respawn_structure_id INT REFERENCES Game_Structure(structure_id),
    nickname             VARCHAR(50),
    character_level      INT DEFAULT 0 NOT NULL CHECK (character_level >= 0 AND character_level <= 100),
    character_health     INT DEFAULT 10,
    character_stamina    INT DEFAULT 10,
    character_class      class_type,
    character_position   DECIMAL(12, 4)
);

CREATE TABLE ItemTemplate (
    item_id          SERIAL PRIMARY KEY,
    item_name        VARCHAR(50) NOT NULL,
    item_type        item_category NOT NULL,
    item_description TEXT,
    item_weight      DECIMAL(5, 2) DEFAULT 0.0,
    item_max_stack   INT CHECK (item_max_stack > 0 AND item_max_stack <= 64),
    rarity           rarity_type NOT NULL,
    item_cost        DECIMAL(10, 2)
);

CREATE TABLE Inventory_Slot (
    slot_id      SERIAL PRIMARY KEY,
    character_id INT REFERENCES Game_Character(character_id),
    item_id      INT REFERENCES ItemTemplate(item_id),
    item_amount  INT DEFAULT 1 CHECK (item_amount > 0),
    is_equipped  BOOLEAN DEFAULT FALSE
);

CREATE TABLE WeaponStas (
    item_id      INT PRIMARY KEY REFERENCES ItemTemplate(item_id),
    damage       DECIMAL(10, 1),
    weapon_bonus bonus_type
);

CREATE TABLE Quest_System (
    quest_id           SERIAL PRIMARY KEY,
    item_id            INT NOT NULL REFERENCES ItemTemplate(item_id),
    quest_name         VARCHAR(50),
    quest_desciription TEXT,
    quest_xp           DECIMAL(10, 2)
);

CREATE TABLE Character_Quests (
    character_id INT REFERENCES Game_Character(character_id),
    quest_id     INT REFERENCES Quest_System(quest_id),
    quest_status quest_status_type NOT NULL,
    PRIMARY KEY (character_id, quest_id)
);

CREATE TABLE Structure_ownership (
    character_id INT REFERENCES Game_Character(character_id),
    structure_id     INT REFERENCES Game_Structure(structure_id),
	structure_role structure_role_type 
);


SQL DML: Вставка тестових даних

Наступний скрипт наповнює кожну таблицю мінімум п'ятьма записами, дотримуючись правильного порядку (спочатку батьківські таблиці, потім дочірні), щоб не порушити цілісність зовнішніх ключів.

TRUNCATE TABLE 
    Account, game_server, Region, Game_Structure, Game_Character, 
    ItemTemplate, Inventory_Slot, WeaponStas, Quest_System, 
    Character_Quests, Structure_ownership 
RESTART IDENTITY CASCADE;
